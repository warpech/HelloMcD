	
	1)	Minimum
		angular.min.js + angular-patch.js
		= angular-angularremote.js

	2)	With Twitter Bootstrap
		jquery.min.js + angular-angularremote.js + bootstrap.js
		= jquery-angular-angularremote-bootstrap.js
		
		bootstrap.min.css
		= jquery-angular-angularremote-bootstrap.css
		
	3)	With Handsontable
		jquery.min.js + angular-angularremote.js + handsontable.js
		= jquery-angular-angularremote-handsontable.js
		
		angular-ui-handsontable.full.css
		= jquery-angular-angularremote-handsontable.css
				
	4)	With Twitter Bootstrap and Handsontable
		jquery-angular-angularremote-bootstrap.js + handsontable.js
		= jquery-angular-angularremote-bootstrap-handsontable.js
		
		jquery-angular-angularremote-bootstrap.css + angular-ui-handsontable.full.css
		= jquery-angular-angularremote-bootstrap-handsontable.css
